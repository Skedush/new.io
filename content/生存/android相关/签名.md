
```zsh
./apksigner verify -v xxx.apk  #查看apk签名信息

apksigner sign -verbose --ks 「jks文件路径」 --v1-signing-enabled [true/false v1打包开启/关闭] --v2-signing-enabled [true/false v2打包开启/关闭] -ks-key-alias [jks别名 key-alias] --ks-pass pass: [jks密码，key store password] --key-pass pass:[密码]

./apksigner sign --v2-signing-enabled true --ks xxx.jks xxx.apk  #对apk使用xxx.jks文件进行签名

jarsigner -sigalg SHA256withRSA -digestalg SHA-256 -keystore [xxx.keystore] -storepass [password] -keypass [password] -verbose -sigfile CERT -signedjar [output.apk] [xxx.apk] [key_alias]
```

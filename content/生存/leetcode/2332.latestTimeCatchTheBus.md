from typing import List


class Solution:
	def latestTimeCatchTheBus(self, buses: List[int], passengers: List[int], capacity: int) -> int:
		buses.sort()
		passengers.sort()
		passed = []
		res = 0
		last_bus = []
		index = 0
		for bus in buses:
			j = 0
			last_bus = []
			while j < capacity and index < len(passengers) and passengers[index] <= bus:
				res = passengers[index]
				passed.append(res)
				last_bus.append(res)
				j += 1
				index += 1

		if len(last_bus) < capacity and res < buses[-1]:

from typing import List

# 
class Solution:
	def getGoodIndices(self, variables: List[List[int]], target: int) -> List[int]:
		good_list = []
		for index, item in enumerate(variables):
			if item[-1] <= target:
				continue
			if (item[0] ** item[1] % 10) ** item[2] % item[3] == target:
				good_list.append(index)
		return good_list
